17 void NestedLoopEquiJoin::executePlanOperation() {
18  // get the two input tables and its hashmaps
19  // First part is the probe table
20  auto left = input.getTable(0);
21  const auto &hleft = input.getTable(1);
22  const auto &pleft = input.getTable(2);
23  const size_t left_size = left->size();
24 
25  // Now fetch the hash table
26  auto right = input.getTable(3);
27  const auto &hright = input.getTable(4);
28  const auto &pright = input.getTable(5);
29  const size_t right_size = right->size();
30 
31  // cast down left hash table, first column contains hashes (value_id_t), second column contains pos_t
32  const auto &lavs0 = hleft->getAttributeVectors(0);
33  const auto &lhvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(lavs0.at(0).attribute_vector);
34  const auto &lavs1 = hleft->getAttributeVectors(1);
35  const auto &lpvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(lavs1.at(0).attribute_vector);
36  // cast down left prefix sum table
37  const auto &lavsp = pleft->getAttributeVectors(0);
38  const auto &lprefixvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(lavsp.at(0).attribute_vector);
39  const size_t lprefixvector_size = lprefixvector->size();
40 
41  // cast down right hash table
42  const auto &ravs0 = hright->getAttributeVectors(0);
43  const auto &rhvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(ravs0.at(0).attribute_vector);
44  const auto &ravs1 = hright->getAttributeVectors(1);
45  const auto &rpvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(ravs1.at(0).attribute_vector);
46  // cast down right first prefix sum table
47  const auto &ravsp = pright->getAttributeVectors(0);
48  const auto &rprefixvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(ravsp.at(0).attribute_vector);
49  const size_t rprefixvector_size = rprefixvector->size();
50 
51  // Prepare mask for right prefix sum
52  auto mask1 = (1 << bits1()) - 1;
53  auto mask2 = ((1 << bits2()) - 1) << bits1();
54  // join
55  size_t left_begin = 0, left_end = 0, right_begin = 0, right_end = 0, partition = 0, rpart = 0;
56  uint32_t lhash;
57  auto lpos_list = new pos_list_t;
58  auto rpos_list = new pos_list_t;
59 
60  lpos_list->reserve(lhvector->size());
61  rpos_list->reserve(lhvector->size());
62 
63  auto multiplier = 1 << bits2();
64  auto shift_right = bits1();
65 
66  // iterate over partitions -> partition gives the offset into the prefix table
67  // The number of partitions depends on the number of bits used for the partitioning
68  for(size_t i = 0, partitions_count = _partitions.size(); i < partitions_count; i++){
69  partition = _partitions[i];
70 
71  // get the corresponding rows for partition using the left prefix table
72  left_begin = lprefixvector->get(0, partition);
73  if(partition == lprefixvector_size - 1) //|| (i+1) == _partitions.size())
74  left_end = left_size;
75  else
76  left_end = lprefixvector->get(0,partition +1);
77 
78  // iterate over values in partition
79  for (size_t left_row = left_begin; left_row < left_end; left_row++) {
80  // find corresponding values of the right relation using the left hash value
81  lhash = lhvector->get(0, left_row);
82 
83  // determine partition in right prefix
84  rpart = ((lhash & mask1) * (multiplier)) + ((lhash & mask2) >> shift_right);
85  right_begin = rprefixvector->get(0, rpart);
86  right_end = rpart < (rprefixvector_size - 1) ? rprefixvector->get(0,rpart +1) : right_size;
87 
88  // iterate over all potentially matching positions in right relation
89  for (size_t right_row = right_begin; right_row < right_end; right_row++) {
90  // compare hashes
91  if (lhash == rhvector->get(0, right_row)) {
92  // remeber matching positions
93  lpos_list->push_back(lpvector->get(0, left_row));
94  rpos_list->push_back(rpvector->get(0, right_row));
95  }
96  }
97  }
98  }
99 
100  // if underlying table is pointer calc, get Actual Table as base for output pointer calc
101  const auto &rp = std::dynamic_pointer_cast<const PointerCalculator>(right);
102  const auto &lp = std::dynamic_pointer_cast<const PointerCalculator>(left);
103 
104  if(lp)
105  left = lp->getActualTable();
106  if(rp)
107  right = rp->getActualTable();
108 
109  // create PointerCalculator and pos_lists for output
110  auto loutput = PointerCalculator::create(left, lpos_list);
111  auto routput = PointerCalculator::create(right, rpos_list);
112 
113  // build output table
114  std::vector<storage::atable_ptr_t > vc;
115  vc.push_back(loutput);
116  vc.push_back(routput);
117 
118  storage::atable_ptr_t result = std::make_shared<storage::MutableVerticalTable>(vc);
119  addResult(result);
120 }
