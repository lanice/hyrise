29 void TableLoad::executePlanOperation() {
30  StorageManager *sm = StorageManager::getInstance();
31  if (!sm->exists(_table_name)) {
32 
33  // Load Raw Table
34  if (_raw) {
35  Loader::params p;
36  p.setHeader(CSVHeader(_file_name));
37  p.setInput(io::RawTableLoader(_file_name));
38  sm->loadTable(_table_name, p);
39 
40  } else if (!_header_string.empty()) {
41  // Load based on header string
42  Loader::params p = Loader::shortcuts::loadWithStringHeaderParams(_file_name, _header_string);
43  sm->loadTable(_table_name, p);
44 
45  } else if (_header_file_name.empty()) {
46  // Load only with single file
47  sm->loadTableFile(_table_name, _file_name);
48 
49  } else if ((!_table_name.empty()) && (!_file_name.empty()) && (!_header_file_name.empty())) {
50  // Load with dedicated header file
51  Loader::params p;
52  p.setCompressed(false);
53  p.setHeader(CSVHeader(_header_file_name));
54  auto params = CSVInput::params().setUnsafe(_unsafe);
55  if (_hasDelimiter)
56  params.setCSVParams(csv::params().setDelimiter(_delimiter.at(0)));
57  p.setInput(CSVInput(_file_name, params));
58  sm->loadTable(_table_name, p);
59  }
60 
61  // We don't load unless the necessary prerequisites are met,
62  // let StorageManager error if table does not exist
63  } else {
64  sm->getTable(_table_name);
65  }
66  auto _table = sm->getTable(_table_name);
67  LOG4CXX_DEBUG(logger, "Loaded Table Size" << _table->size());
68  addResult(_table);
69 }
