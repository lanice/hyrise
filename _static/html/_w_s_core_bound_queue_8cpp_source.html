112 std::vector<std::shared_ptr<Task> > WSCoreBoundQueue::stopQueue() {
113  if (_status != STOPPED) {
114  // the thread to be stopped is either executing a task, or waits for the condition variable
115  // set status to "TO_STOP" so that the thread either quits after executing the task, or after having been notified by the condition variable
116  // we need the mutex here, otherwise, we might call notify prior to the thread going to sleep
117  {
118  std::lock_guard<std::mutex> lk(_queueMutex);
119  _status = TO_STOP;
120  //wake up thread in case thread is sleeping
121  _condition.notify_one();
122  }
123  _thread->join();
124  delete _thread;
125  _thread = NULL;
126  _status = STOPPED;
127  }
