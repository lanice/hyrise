101 std::shared_ptr<PlanOperation> LayoutSingleTable::parse(Json::Value &data) {
102  std::shared_ptr<LayoutSingleTable> s = std::make_shared<LayoutSingleTable>();
103  s->setNumRows(data["num_rows"].asUInt());
104 
105  if (data.isMember("layouter")) {
106  if (data["layouter"] == "BaseLayouter")
107  s->setLayouter(BaseLayouter);
108  else if (data["layouter"] == "CandidateLayouter")
109  s->setLayouter(CandidateLayouter);
110  else if (data["layouter"] == "DivideAndConquerLayouter")
111  s->setLayouter(DivideAndConquerLayouter);
112  else
113  throw std::runtime_error("Layouter not available, chose different implementation");
114  } else {
115  s->setLayouter(BaseLayouter);
116  }
117 
118  // Parse attributes
119  if (data["attributes"].isArray()) {
120  for (unsigned i = 0; i < data["attributes"].size(); ++i) {
121  s->addFieldName(data["attributes"][i].asString());
122  }
123  } else {
124  throw QueryParserException("Parse LayoutSingleTable failed: no attributes defined");
125  }
126 
127  // Parse queries
128  if (data["operators"].isArray()) {
129  Json::Value ops = data["operators"];
130  for (unsigned i = 0; i < ops.size(); ++i) {
131  Json::Value q = ops[i];
132  BaseQuery query;
133  query.selectivity = q["selectivity"].asDouble();
134  query.weight = q["weight"].asUInt64();
135 
136  Json::Value a = q["attributes"];
137  for (unsigned j = 0; j < a.size(); ++j) {
138  std::string name = a[j].asString();
139  // find name in _names to get the position
140  for (unsigned k = 0; k < s->_names.size(); ++k)
141  if (name.compare(s->_names[k]) == 0)
142  query.positions.push_back(k);
143  }
144 
145  // Add query to layouter
146  s->addQuery(query);
147  }
148  } else {
149  throw QueryParserException("Parse LayoutSingleTable failed: no queries defined");
150  }
151 
152  // TODO from JSON Parsing
153  if (data.isMember("max_results"))
154  s->setMaxResults(data["max_results"].asUInt());
155  else
156  s->setMaxResults(10);
157 
158  return s;
