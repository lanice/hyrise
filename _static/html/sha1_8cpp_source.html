192 void SHA1Update(SHA1_CTX *context, const unsigned char *data, u_int32_t len) {
193  u_int32_t i;
194  u_int32_t j;
195 
196  j = context->count[0];
197  if ((context->count[0] += len << 3) < j)
198  context->count[1]++;
199  context->count[1] += (len >> 29);
200  j = (j >> 3) & 63;
201  if ((j + len) > 63) {
202  memcpy(&context->buffer[j], data, (i = 64 - j));
203  SHA1Transform(context->state, context->buffer);
204  for (; i + 63 < len; i += 64) {
205  SHA1Transform(context->state, &data[i]);
206  }
207  j = 0;
208  } else i = 0;
209  memcpy(&context->buffer[j], &data[i], len - i);
210 }
