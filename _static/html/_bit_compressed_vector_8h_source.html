72  T get(size_t column, size_t row) const {
73  checkAccess(column, row);
74 
75  T result;
76  auto offset = _blockOffset(row);
77  auto colOffset = _offsetForColumn(column);
78  auto block = _blockPosition(row) + (offset + colOffset) / _bit_width;
79 
80  // Adjust the column offset to the new block
81  colOffset = (offset + colOffset) % _bit_width;
82 
83  uint64_t bounds = _bit_width - colOffset;
84  uint64_t baseMask = (1ull << _bits[column]) - 1ull;
85 
86  result = (_data[block] >> colOffset) & baseMask;
87 
88  if (bounds < _bits[column]) {
89  colOffset = _bits[column] - bounds;
90  baseMask = (1ull << colOffset) - 1ull;
91  result |= (baseMask & _data[block + 1]) << bounds;
92  }
93  return result;
94  }

96  void set(size_t column, size_t row, T value) {
97  checkAccess(column, row);
98 
99  auto offset = _blockOffset(row);
100  auto colOffset = _offsetForColumn(column);
101  auto block = _blockPosition(row) + (offset + colOffset) / _bit_width;
102 
103  // Adjust the column offset to the new block
104  colOffset = (offset + colOffset) % _bit_width;
105 
106  uint64_t bounds = _bit_width - colOffset;
107  uint64_t baseMask = (1ull << _bits[column]) - 1ull;
108  uint64_t mask = ~(baseMask << colOffset);
109 
110  _data[block] &= mask;
111  _data[block] |= ((uint64_t) value & baseMask) << colOffset;
112 
113  if (bounds < _bits[column]) {
114  mask = ~(baseMask >> bounds);
115  _data[block + 1] &= mask;
116  _data[block + 1] |= ((uint64_t) value & baseMask) >> bounds;
117  }

120  /*
121  Reserve memory for the given number of rows. memory will only be
122  allocated if the number of rows requires a larger number of blocks
123  than before and it can only be incremented.
124  */
125  void reserve(size_t rows) {
126  if (rows > 0 && _blocks(rows) > _blocks(_allocatedBlocks)) {
127  // Allocate the new memory, copy the old memory and swap the
128  // data
129  assert(_blocks(rows) > 0);
130  storage_t *newMemory = _allocate(_blocks(rows));
131  if (_allocatedBlocks > 0)
132  std::memcpy(newMemory, _data, _allocatedBlocks * sizeof(storage_t));
133 
134  std::swap(_data, newMemory);
135 
136  // Only deallocate if there was something allocated
137  if (newMemory != nullptr)
138  Strategy::deallocate(newMemory, _allocatedBlocks * sizeof(storage_t));
139 
140  // set new allocarted blocks
141  _allocatedBlocks = _blocks(rows);
142  }
143  }
