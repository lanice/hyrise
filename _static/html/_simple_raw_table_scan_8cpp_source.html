71 void SimpleRawTableScan::executePlanOperation() {
72  auto table = std::dynamic_pointer_cast<const RawTable>(input.getTable(0));
73  if (!table)
74  throw std::runtime_error("Input table is no uncompressed raw table");
75 
76  // Prepare a result that contains the result semi-compressed, and only row-wise
77  metadata_list meta(table->columnCount());
78  for(size_t i=0; i<table->columnCount(); ++i)
79  meta[i] = table->metadataAt(i);
80  auto result = std::make_shared<Table>(&meta,
81  nullptr,
82  1, /* initial size */
83  false, /* sorted */
84  false /* compressed */);
85 
86  // Prepare the copy operator
87  storage::copy_value_functor_raw_table fun(result, table);
88  storage::type_switch<hyrise_basic_types> ts;
89  auto positions = new storage::pos_list_t;
90 
91  size_t tabSize = table->size();
92  for(size_t row=0; row < tabSize; ++row) {
93  if ((*_comparator)(row)) {
94  if (_materializing) {
95  result->resize(result->size() + 1);
96  for(size_t i=0; i < result->columnCount(); ++i) {
97  fun.setValues(result->size() - 1, row, i);
98  ts(table->typeOfColumn(i), fun);
99  }
100  }
101  else {
102  positions->push_back(row);
103  }
104  }
105  }
106 
107  if (_materializing) {

109  } else {
