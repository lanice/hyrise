86 void PlanOperation::computeDeferredIndexes() {
87  _field_definition = _indexed_field_definition;
88  if (!_named_field_definition.empty()) {
89  assert(_field_definition.empty());
90  const auto& table = input.getTable(0);
91  if ((_named_field_definition.size() == 1) && (_named_field_definition[0] == "*")) {
92  _field_definition.resize(table->columnCount());
93  std::iota(std::begin(_field_definition), std::end(_field_definition), 0);
94  } else {
95  for (size_t i = 0; i < _named_field_definition.size(); ++i) {
96  _field_definition.push_back(table->numberOfColumn(_named_field_definition[i]));
97  }
98  }
99  }
100  assert(_field_definition.size() >= (_indexed_field_definition.size() + _named_field_definition.size()));
101 }
