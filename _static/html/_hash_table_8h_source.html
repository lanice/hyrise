72 template<class T>
73 class GroupKeyHash {
74 public:
75  size_t operator()(const T &key) const {
76  static auto hasher = std::hash<value_id_t>();
77 
78  std::size_t seed = 0;
79  for (size_t i = 0, key_size = key.size();
80  i < key_size; ++i) {
81  // compare boost hash_combine
82  seed ^= hasher(key[i]) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
83  }
84  return seed;
85  }

87  static T getGroupKey(const hyrise::storage::c_atable_ptr_t &table,
88  const field_list_t &columns,

92  T key;
