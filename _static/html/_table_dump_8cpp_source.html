118 void SimpleTableDump::dumpDictionary(std::string name, std::shared_ptr<AbstractTable> table, size_t col) {
119  std::string fullPath = _baseDirectory + "/" + name + "/" + table->nameOfColumn(col) + ".dict.dat";
120  std::ofstream data (fullPath, std::ios::out | std::ios::binary);
121 
122  // We make a small hack here, first we obtain the size of the
123  // dictionary then we virtually create all value ids, this can break
124  // if the dictionary has no contigous value ids
125  size_t dictionarySize = table->dictionaryAt(col)->size();
126  write_to_stream_functor fun(data, table);
127  type_switch<hyrise_basic_types> ts;
128  for(size_t i=0; i < dictionarySize; ++i) {
129  fun.setCol(col);
130  fun.setVid(i);
131  ts(table->typeOfColumn(col), fun);
132  }
133  data.close();
134 }

147 void SimpleTableDump::dumpHeader(std::string name, std::shared_ptr<AbstractTable> table) {
148  std::stringstream header;
149  std::vector<std::string> names, types;
150  std::vector<uint32_t> parts;
151 
152  // Get names and types
153  for(size_t i=0; i < table->columnCount(); ++i) {
154  names.push_back(table->nameOfColumn(i));
155  types.push_back(data_type_to_string(table->typeOfColumn(i)));
156  }
157 
158  // This calculation will break if the width of the value_id changes
159  // or someone forgets to simply update the width accordingly in the
160  // constructor of the table
161  for(size_t i=0; i < table->partitionCount(); ++i) {
162  parts.push_back(table->partitionWidth(i));
163  }
164 
165  // Dump and join
166  header << std::accumulate(names.begin(), names.end(), std::string(), infix(" | ")) << "\n";
167  header << std::accumulate(types.begin(), types.end(), std::string(), infix(" | ")) << "\n";
168  std::vector<std::string> allParts;
169  for(size_t i=0; i < parts.size(); ++i) {
170  auto p = parts[i];
171  auto tmp = std::vector<std::string>(p, std::to_string(i) + "_R");
172  allParts.insert(allParts.end(), tmp.begin(), tmp.end());
173  }
174  header << std::accumulate(allParts.begin(), allParts.end(), std::string(), infix(" | ")) << "\n";
175  header << "===";
176 
177  std::string fullPath = _baseDirectory + "/" + name + "/header.dat";
178  std::ofstream data (fullPath, std::ios::out | std::ios::binary);
179  data << header.str();
180  data.close();
181 }

198 bool SimpleTableDump::dump(std::string name, std::shared_ptr<AbstractTable> table) {
199  verify(table);
200  auto mainTable = std::dynamic_pointer_cast<Store>(table)->getMainTables()[0];
201  prepare(name);
202  for(size_t i=0; i < mainTable->columnCount(); ++i) {
203  // For each attribute dump dictionary and values
204  dumpDictionary(name, mainTable, i);
205  dumpAttribute(name, mainTable, i);
206  }
207 
208  dumpHeader(name, mainTable);
209  dumpMetaData(name, mainTable);
210
