31 void PrettyPrinter::special_print(T& input, std::ostream& outStream, const std::string tableName, const size_t& limit, const size_t& start) {
32  ftprinter::FTPrinter tp(tableName, outStream);
33  tp.addColumn("#rowid", 6);
34  const size_t columns = input->columnCount();
35  for (size_t column_index = 0; column_index < columns; ++column_index) {
36  // Auto adjusting widths means iterating over the table twice, but we use it for
37  // debugging purposes only, anyways, so we'll go with beauty of output here
38  auto name = input->nameOfColumn(column_index);
39  size_t width = std::accumulate(RangeIter(0), RangeIter(input->size()),
40  // minimum width is 4
41  name.size() > 4 ? name.size() : 4,
42  [&] (size_t max, const size_t& row) -> size_t {
43  size_t sz = generateValue(input, column_index, row).size();
44  return sz > max ? sz : max;
45  });
46  tp.addColumn(name, width);
47  }
48 
49  if (limit < (size_t) - 1) {
50  outStream << "(showing first " << limit << " rows)" << std::endl;
51  }
52 
53  if (tableName.size() > 0)
54  tp.printTableName();
55  tp.printHeader();
56  for (size_t row = start; row < input->size() && row < limit; ++row) {
57  tp << row;
58  for (size_t column = 0; column < columns; ++column) {
59  tp << generateValue(input, column, row);
60  }
61  }
62  tp.printFooter();
63 }
