53  std::vector<pos_t>* sort() const {
54  std::vector<pair_t> result;
55 
56  for (size_t row = 0; row < _t->size(); ++row) {
57  result.push_back({ExtractFunctor<T>::extractValue(_t, _f, row), row});
58  }
59 
60  auto asc_sort = [](const pair_t& left, const pair_t& right) {
61  return (left.value < right.value);
62  };
63 
64  auto desc_sort = [](const pair_t& left, const pair_t& right) {
65  return (left.value > right.value);
66  };
67 
68 
69  std::stable_sort(result.begin(),
70  result.end(),
71  _asc ? asc_sort : desc_sort
72  );
73 
74  auto r = new std::vector<pos_t>;
75  r->reserve(result.size());
76  for (const pair_t& p: result)
77  r->push_back(p.row);

90 void SortScan::executePlanOperation() {
91  const auto& table = input.getTable(0);
92  // Sorted Position List
93  std::vector<pos_t> *sorted_pos;
94 
95  // When table is not only a table but also using an ordered dictionary on sort field,
96  // we can just sort by value_id
97  // TODO: fix Table<> template
98  auto base_table = std::dynamic_pointer_cast<const Table>(table);
99  if ((table->dictionaryAt(_sort_field)->isOrdered()) && (base_table)) {
100  sorted_pos = ColumnSorter<ValueId, ExtractValueId>(table, _sort_field, asc).sort();
101  } else {
102  switch (table->metadataAt(_sort_field)->getType()) {
103  case IntegerType:
104  sorted_pos = ColumnSorter<hyrise_int_t, ExtractValue>(table, _sort_field, asc).sort();
105  break;
106  case FloatType:
107  sorted_pos = ColumnSorter<hyrise_float_t, ExtractValue>(table, _sort_field, asc).sort();
108  break;
109  case StringType:
110  sorted_pos = ColumnSorter<hyrise_string_t, ExtractValue>(table, _sort_field, asc).sort();
111  break;
112  default:
113  throw std::runtime_error("Datatype not supported");
114  }
115  }
116 
117  storage::atable_ptr_t result;
118 
119  if (producesPositions) {
120  result = PointerCalculator::create(table, sorted_pos);
121  } else {
122  result = table->copy_structure_modifiable(nullptr, true);
123  size_t result_row = 0;
124  for (const auto& p: *sorted_pos) {
125  result->copyRowFrom(table, p, result_row++);
126  }
127  delete sorted_pos;
128  }
129  addResult(result);
130 }
