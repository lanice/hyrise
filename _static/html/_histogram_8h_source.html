66 template<typename T>
67 void Histogram::executeHistogram() {
68  const auto &tab = getInputTable();
69  const auto tableSize = getInputTable()->size();
70  const auto field = _field_definition[0];
71 
72  //Prepare mask
73  auto mask = ((1 << bits()) - 1) << significantOffset();
74 
75  // Prepare Output Table
76  auto result = createOutputTable(1 << bits());
77  auto pair = getDataVector(result);
78 
79  // Iterate and hash based on the part description
80  size_t start=0, stop=tableSize;
81  if (_count > 0) {
82  start = (tableSize / _count) * _part;
83  stop = (_count -1) == _part ? tableSize : (tableSize/_count) * (_part + 1);
84  }
85 
86  // check if tab is PointerCalculator; if yes, get underlying table and actual rows and columns
87  auto p = std::dynamic_pointer_cast<const PointerCalculator>(tab);
88  if (p) {
89  auto ipair = getDataVector(p->getActualTable(), p->getTableColumnForColumn(field));
90  const auto &ivec = ipair.first;
91  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(p->getTableColumnForColumn(field)));
92  const auto &offset = ipair.second;
93 
94  auto hasher = std::hash<T>();
95  for(size_t row = start; row < stop; ++row) {
96  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, p->getTableRowForRow(row))));
97  pair.first->inc(0, (hash_value & mask) >> significantOffset());
98  }
99  } else {
100 
101  // output of radix join is MutableVerticalTable of PointerCalculators
102  auto mvt = std::dynamic_pointer_cast<const storage::MutableVerticalTable>(tab);
103  if(mvt){
104  auto pc = mvt->containerAt(field);
105  auto p = std::dynamic_pointer_cast<const PointerCalculator>(pc);
106  if(p){
107  auto ipair = getDataVector(p->getActualTable(), p->getTableColumnForColumn(field));
108  const auto &ivec = ipair.first;
109  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(p->getTableColumnForColumn(field)));
110  const auto &offset = ipair.second;
111 
112  auto hasher = std::hash<T>();
113  for(size_t row = start; row < stop; ++row) {
114  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, p->getTableRowForRow(row))));
115  pair.first->inc(0, (hash_value & mask) >> significantOffset());
116  }
117  } else {
118  throw std::runtime_error("Histogram only supports MutableVerticalTable of PointerCalculators; found other AbstractTable than PointerCalculator inside od MutableVerticalTable.");
119  }
120  } else {
121  // else; we expect a raw table
122  auto ipair = getDataVector(tab, field);
123  const auto &ivec = ipair.first;
124  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(field));
125  const auto &offset = ipair.second;
126 
127  auto hasher = std::hash<T>();
128  for(size_t row = start; row < stop; ++row) {
129  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, row)));
130  pair.first->inc(0, (hash_value & mask) >> significantOffset());
131  }
132  }
133  }
134  addResult(result);
135 }
