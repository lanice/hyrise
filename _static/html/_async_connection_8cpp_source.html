105 void write_cb(struct ev_loop *loop, struct ev_async *w, int revents) {
106  AsyncConnection *conn = (AsyncConnection *) w->data;
107

111  method = (char *)"GET";

113  case EBB_POST:
114  method = (char *)"POST";

117  break;
118  }

124  struct tm *timeinfo;
125  char timestr[80];
126  time(&rawtime);
127  timeinfo = localtime(&rawtime);
128  strftime(timestr, sizeof(timestr), "%Y-%m-%d %H:%M:%S %z", timeinfo);
129 
130  if (conn->connection != nullptr) {
131  conn->write_buffer = (char *)malloc(max_header_length + conn->response_length);
132  conn->write_buffer_len = 0;
133 
134  // Copy the http status code
135  conn->code = conn->code == 0 ? 200 : conn->code;
136  conn->contentType = conn->contentType.size() == 0 ? "application/json" : conn->contentType;
137  conn->write_buffer_len += snprintf((char *)conn->write_buffer, max_header_length,
138  "HTTP/1.1 %lu OK\r\nContent-Type: %s\r\nContent-Length: %lu\r\nConnection: close\r\n\r\n",
139  conn->code,
140  conn->contentType.c_str(),
141  conn->response_length);
142 
143  // Append the response
144  memcpy(conn->write_buffer + conn->write_buffer_len, conn->response, conn->response_length);
145  conn->write_buffer_len += conn->response_length;
146  ebb_connection_write(conn->connection, conn->write_buffer, conn->write_buffer_len, continue_responding);
147  // We need to wait for `continue_responding` to fire to be sure the client has been sent all the data
148  printf("%s [%s] %s %s (%f s)\n", inet_ntoa(conn->addr.sin_addr), timestr, method, conn->path, duration);
149  } else {
150  printf("%s [%s] %s %s (%f s) not sent\n", inet_ntoa(conn->addr.sin_addr), timestr, method, conn->path, duration);
151  }
152  ev_async_stop(conn->ev_loop, &conn->ev_write);
153  // When connection is nullptr, `continue_responding` won't fire since we never sent data to the client,
154  // thus, we'll need to clean up manually here, while connection has already been cleaned up in on `on_response`
155  if (conn->connection == nullptr) delete conn;
