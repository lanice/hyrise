68 void PointerCalculator::updateFieldMapping() {
69  slice_for_slice.clear();
70  offset_in_slice.clear();
71  width_for_slice.clear();
72  slice_count = 0;
73 
74  size_t field = 0, dst_field = 0;
75  size_t last_field = 0;
76  bool last_field_set = false;
77 
78  for (size_t src_slice = 0; src_slice < table->partitionCount(); src_slice++) {
79  last_field_set = false;
80  for (size_t src_field = 0; src_field < table->partitionWidth(src_slice); src_field++) {
81  // Check if we have to increase the fields until we reach
82  // a projected attribute
83  if (fields && (fields->size() <= dst_field || fields->at(dst_field) != field)) {
84  field++;
85  continue;
86  }
87 
88  if (!last_field_set || field > last_field + 1) { // new slice
89  slice_count++;
90  slice_for_slice.push_back(src_slice);
91  offset_in_slice.push_back(src_field);
92  width_for_slice.push_back(1);
93  } else {
94  width_for_slice[slice_count - 1] += 1;
95  }
96 
97  dst_field++;
98  last_field = field;
99 
100  if (!last_field_set) {
101  last_field_set = true;
102  }
103 
104  field++;
105  }
106  }
107 }

364  auto result = new pos_list_t;
