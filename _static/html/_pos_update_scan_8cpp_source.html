31 void PosUpdateScan::executePlanOperation() {
32  auto c_pc = checked_pointer_cast<const PointerCalculator>(input.getTable(0));
33  auto c_store = checked_pointer_cast<const storage::Store>(c_pc->getActualTable());
34 
35  // Cast the constness away
36  auto store = std::const_pointer_cast<storage::Store>(c_store);
37 
38  // Get the current maximum size
39  const auto& beforSize = store->size();
40 
41  // Get the offset for inserts into the delta and the size of the delta that
42  // we need to increase by the positions we are inserting
43  auto writeArea = store->appendToDelta(c_pc->getPositions()->size());
44 
45  // Get the modification record for the current transaction
46  auto& txmgr = tx::TransactionManager::getInstance();
47  auto& modRecord = txmgr[_txContext.tid];
48 
49  // Functor we use for updating the data
50  set_json_value_functor fun(store->getDeltaTable());
51  storage::type_switch<hyrise_basic_types> ts;
52 
53  size_t counter = 0;
54  for(const auto& p : *(c_pc->getPositions())) {
55  // First delete the old record
56  bool deleteOk = store->markForDeletion(p, _txContext.tid) == hyrise::tx::TX_CODE::TX_OK;
57  if(!deleteOk) {
58  txmgr.abort();
59  throw std::runtime_error("Aborted TX because TID of other TX found");
60  }
61  modRecord.deletePos(store, p);
62  //store->setTid(p, _txContext.tid);
63 
64  // Copy the old row from the main
65  store->copyRowToDelta(store, p, writeArea.first+counter, _txContext.tid);
66  // Update all the necessary values
67  for(const auto& kv : _raw_data) {
68  const auto& fld = store->numberOfColumn(kv.first);
69  fun.set(fld, writeArea.first+counter, kv.second);
70  ts(store->typeOfColumn(fld), fun);
71  }
72 
73  // Insert the new one
74  modRecord.insertPos(store, beforSize+counter);
75  ++counter;
76  }
77 
78  // Update affected rows
79  auto rsp = getResponseTask();
80  if (rsp != nullptr)
81  rsp->incAffectedRows(counter);
82 
83  addResult(c_store);
84 }
