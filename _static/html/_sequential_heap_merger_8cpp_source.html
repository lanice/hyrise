117 AbstractTable::SharedDictionaryPtr SequentialHeapMerger::createNewDict(const std::vector<hyrise::storage::c_atable_ptr_t > &input_tables,
118  std::vector<AbstractTable::SharedDictionaryPtr > &value_id_maps,
119  std::vector<std::vector<value_id_t> > &value_id_mapping,
120  size_t column_index,
121  bool useValid,
122  const std::vector<bool>& valid) {
123  // Heap Queue
124  std::priority_queue<DictMergeHelper<T>> queue;
125 
126  value_id_mapping.resize(input_tables.size());
127 
128  size_t part_counter = 0;
129  for(size_t p=0, stop=input_tables.size(); p < stop; ++p) {
130 
131  auto dict = std::dynamic_pointer_cast<BaseDictionary<T>>(value_id_maps[p]);
132  value_id_mapping[p].resize(value_id_maps[p]->size());
133 
134  DictMergeHelper<T> helper;
135  helper.it = dict->begin();
136  helper.end = dict->end();
137  helper.tab_index = p;
138 
139  if (useValid) {
140  helper.ref.resize(dict->size(), false);
141  for(size_t i=0, tabs = input_tables[p]->size(); i < tabs; ++i) {
142  if (valid[part_counter + i]) {
143  helper.ref[input_tables[p]->getValueId(column_index, i).valueId] = true;
144  }
145  }
146  } else {
147  helper.ref.resize(dict->size(), true);
148  }
149 
150  if (!helper.done()) {
151  queue.push(helper);
152  }
153 
154  // Update the row counter
155  part_counter += input_tables[p]->size();
156  }
157 
158  // Create new Dict
159  bool first = true;
160  bool assigned = false;
161 
162  T last_value;
163  auto new_dict = std::make_shared<OrderPreservingDictionary<T>>( hyrise::functional::sum(value_id_mapping, 0ul, [](std::vector<value_id_t>& v){ return v.size(); }));
164  while(!queue.empty()) {
165 
166  auto element = queue.top();
167 
168  T curr_val = *(element.it);
169  queue.pop();
170 
171  if (first || !assigned || (last_value != curr_val)) {
172  if (element.valid()) {
173  new_dict->addValue(curr_val);
174  }
175  }
176 
177  if (element.valid()) {
178  value_id_mapping[element.tab_index][element.it.getValueId()] = new_dict->size() - 1;
179  last_value = curr_val;
180  assigned = true;
181  }
182 
183  element.next();
184  if (!element.done()) {
185  queue.push(element);
186  }
187  first = false;
188  }
189 
190  new_dict->shrink();
191  return new_dict;
192 }

195  size_t source_column_index,
196  hyrise::storage::atable_ptr_t &merged_table,

202 
203  // copy all value ids to the new doc vector

207  size_t part_counter = 0;

209  for (size_t row = 0; row < input_tables[table]->size(); row++) {
