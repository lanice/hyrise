120  //creating fields from grouping fields

122  //creating fields from aggregate functions
123  for (const auto & fun: _aggregate_functions) {
124  ColumnMetadata *m = new ColumnMetadata(fun->columnName(getInputTable(0)->nameOfColumn(fun->getField())), fun->getType());

136  return agg_tab;

179 template<typename HashTableType, typename MapType, typename KeyType>
180 void GroupByScan::executeGroupBy() {
181  auto resultTab = createResultTableLayout();
182 
183  auto groupResults = getInputHashTable();
184  // Allocate some memory for the result tab and resize the table
185  resultTab->resize(groupResults->numKeys());
186 
187  pos_t row = 0;
188  typename HashTableType::map_const_iterator_t it1, it2, end;
189  // set iterators: in the sequential case, getInputTable() returns an AggregateHashTable, in the parallel case a HashTableView<>
190  // Alternatively, a common type could be introduced
191  if (_count < 1) {
192  auto aggregateHashTable = std::dynamic_pointer_cast<const HashTableType>(groupResults);
193  it1 = aggregateHashTable->getMapBegin();
194  end = aggregateHashTable->getMapEnd();
195  } else {
196  auto hashTableView = std::dynamic_pointer_cast<const HashTableView<MapType, KeyType> >(groupResults);
197  it1 = hashTableView->getMapBegin();
198  end = hashTableView->getMapEnd();
199  }
200  for (it2 = it1; it1 != end; it1 = it2) {
201  // outer loop over unique keys
202  auto pos_list = std::make_shared<pos_list_t>();
203  for (; (it2 != end) && (it1->first == it2->first); ++it2) {
204  // inner loop, all keys equal to it1->first
205  pos_list->push_back(it2->second);
206  }
207  writeGroupResult(resultTab, pos_list, row);
208  row++;
209  }
210 
211  this->addResult(resultTab);
212 }
