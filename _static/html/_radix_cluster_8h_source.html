63 
64  // Prepare mask
65  auto mask = ((1 << bits()) - 1) << significantOffset();
66 
67  // Cast the vectors to the lowest part in the hierarchy
68  const auto &data_hash = getDataVector(result).first;
69  const auto &data_pos = getDataVector(result, 1).first;
70 
71  // Calculate start stop
72  _start = 0; _stop = tableSize;
73  if (_count > 0) {
74  _start = (tableSize / _count) * _part;
75  _stop = (_count -1) == _part ? tableSize : (tableSize/_count) * (_part + 1);
76  }
77 
78  // check if tab is PointerCalculator; if yes, get underlying table and actual rows and columns

80  if (p) {
81  auto ipair = getDataVector(p->getActualTable());
82  const auto &ivec = ipair.first;

85  const auto &offset = p->getTableColumnForColumn(field) + ipair.second;
86

88  for(decltype(tableSize) row = _start; row < _stop; ++row) {
89  // Calculate and increment the position

91  auto offset = (hash_value & mask) >> _significantOffset;
92  auto pos_to_write = data_prefix_sum->inc(0, offset);
93 
94  // Perform the clustering
95  data_hash->set(0, pos_to_write, hash_value);
96  data_pos->set(0, pos_to_write, p->getTableRowForRow(row));
97  }
98  } else {
99

101  auto mvt = std::dynamic_pointer_cast<const storage::MutableVerticalTable>(tab);
102  if(mvt){
103

105  auto p = std::dynamic_pointer_cast<const PointerCalculator>(pc);
106  if(p){
107  auto ipair = getDataVector(p->getActualTable());
108  const auto &ivec = ipair.first;
109

111  const auto &offset = p->getTableColumnForColumn(field) + ipair.second;

117  auto offset = (hash_value & mask) >> _significantOffset;
118  auto pos_to_write = data_prefix_sum->inc(0, offset);
119

123  }

127  }
