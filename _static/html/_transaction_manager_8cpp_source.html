192 transaction_cid_t TransactionManager::commitTransaction(transaction_id_t tid) {
193  if (!isRunningTransaction(tid)) {
194  throw std::runtime_error("Transaction is not currently running");
195  }
196  auto& txmgr = getInstance();
197  auto& tx_data = getTransactionData(tid);
198  auto& _txContext = tx_data._context;
199  const auto& modifications = tx_data._modifications;
200 
201  _txContext.cid = txmgr.prepareCommit();
202 
203  // Only update the required positions
204  for (auto& kv: modifications.deleted) {
205  auto weak_table = kv.first;
206  // Only deleted records have to be checked for validity as newly inserted
207  // records will be always only written by us
208  if (auto store = getStore(weak_table.lock())) {
209  if (tx::TX_CODE::TX_OK != store->checkForConcurrentCommit(kv.second, tid)) {
210  txmgr.abort();
211  throw std::runtime_error("Aborted TX with Last Commit ID != New Commit ID");
212  }
213  }
214  }
215 
216  for (auto& kv: modifications.inserted) {
217  auto weak_table = kv.first;
218  if (auto store = getStore(weak_table.lock())) {
219  auto result = store->commitPositions(kv.second, _txContext.cid, true);
220  if (result != TX_CODE::TX_OK) {
221  txmgr.abort();
222  throw std::runtime_error("Aborted TX with "); // TODO at return code to error message
223  }
224  }
225  }
226 
227  for (auto& kv: modifications.deleted) {
228  auto weak_table = kv.first;
229  if (auto store = getStore(weak_table.lock())) {
