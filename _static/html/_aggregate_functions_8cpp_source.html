103 AggregateFun *parseAggregateFunction(const Json::Value &data) {
104  int ftype = -1;
105  if (data["type"].isNumeric()) {
106  ftype = data["type"].asUInt();
107  } else if (data["type"].isString()) {
108  ftype = getAggregateFunctionMap()[data["type"].asString()];
109  }
110  switch (ftype) {
111  case AggregateFunctions::SUM:
112  return SumAggregateFun::parse(data);
113  case AggregateFunctions::COUNT:
114  return CountAggregateFun::parse(data);
115  case AggregateFunctions::AVG:
116  return AverageAggregateFun::parse(data);
117  default:
118  throw std::runtime_error("Aggregation function not supported in GroupByScan");
119  }
