30 void TableGenerator::prepare(size_t p) {
31  if (p <= _prepareSize)
32  return;
33 
34  if (_prepareSize > 0) {
35  _values.clear();
36  }
37 
38  _prepareSize = p;
39 
40  std::mt19937 gen;
41  std::uniform_int_distribution<int64_t> dist(0, std::numeric_limits<int64_t>::max());
42 
43  // First, generate all random numbers we need, this is blazing
44  // fast
45  std::vector<int64_t> v;
46 
47  // Depending on the cycle of the random number generator, we will
48  // need to generate more random numbers than we actually need to
49  // make sure we have still enough without double entries
50  int upper = (_prepareSize * 105) / 100;
51  for (int i = 0; i < upper; ++i) {
52  v.push_back(dist(gen));
53  }
54 
55  // Now, sort the vector
56  std::sort(v.begin(), v.end());
57  std::set<int64_t>::iterator it = _values.begin();
58 
59  // Insert the elements from the vector sorted to the set and ta da
60  // we are fast
61  for (size_t i = 0; i < v.size(); ++i) {
62  it = _values.insert(it, v[i]);
63  }
64 
65  // If we have not enough distinct values we will need to generate
66  // more random values until we are done, but this should be only a
67  // small amount.
68  while (_values.size() < _prepareSize) {
69  _values.insert(dist(gen));
70  }
71 
72 }

196 std::vector<int64_t> *TableGenerator::create_dicts(size_t dict_size_main, size_t dict_size_delta, size_t intersection, size_t tail) {
197  Progress p(dict_size_main + dict_size_delta);
198 
199  std::vector<int64_t> *v = new std::vector<int64_t>;
200  v->reserve(dict_size_main + dict_size_delta);
201 
202  // maindict
203  for (size_t i = 0; i < dict_size_main; ++i) {
204  v->push_back(dict_size_delta + i);
205  p.tick();
206  }
207 
208  // deltadict
209  size_t head = dict_size_delta - intersection - tail;
210 
211  for (size_t i = 0; i < head; ++i) {
212  v->push_back(i);
213  p.tick();
214  }
215 
216  for (size_t i = head; i < head + intersection; ++i) {
217  v->push_back(v->at(i - head));
218  p.tick();
219  }
220 
221  for (size_t i = head + intersection; i < dict_size_delta; ++i) {
222  v->push_back(dict_size_main + dict_size_delta + i);
223  p.tick();
224  }
225 
226  v->at(dict_size_main + dict_size_delta - tail - 1) = v->at(dict_size_main - 1);
227 
228  return v;

874  size_t x = row_count / distinct_count;
875  std::vector<hyrise::storage::c_atable_ptr_t > tables;
876  const auto& base_table = int_random(distinct_count, column_count);
877  tables.push_back(base_table);
878 
879  for (size_t i = 0; i <= x; i++) {
880  tables.push_back(base_table);
881  }
882 
883  size_t fill_up = row_count - x * distinct_count;
884 
885  if (fill_up > 0) {
886  const auto& fill_up_table = int_random(fill_up, column_count);
887  tables.push_back(fill_up_table);
888  }
889 
890  const auto& result = TableMerger(new DefaultMergeStrategy(), new SequentialHeapMerger()).merge(tables);
891  return result[0];
