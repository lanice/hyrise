40 void Store::merge() {
41  if (merger == nullptr) {
42  throw std::runtime_error("No Merger set.");
43  }
44 
45  // Create new delta and merge
46  atable_ptr_t new_delta = delta->copy_structure_modifiable();
47 
48  // Prepare the merge
49  std::vector<c_atable_ptr_t> tmp(main_tables.begin(), main_tables.end());
50  tmp.push_back(delta);
51 
52  // get valid positions
53  std::vector<bool> validPositions(_cidBeginVector.size());
54  functional::forEachWithIndex(_cidBeginVector, [&](size_t i, bool v){
55  validPositions[i] = isVisibleForTransaction(i, tx::TransactionManager::getInstance().getLastCommitId(), tx::MERGE_TID);
56  });
57 
58  main_tables = merger->merge(tmp, true, validPositions);
59 
60  // Fixup the cid and tid vectors
61  _cidBeginVector = std::vector<tx::transaction_cid_t>(main_tables[0]->size(), tx::UNKNOWN_CID);
62  _cidEndVector = std::vector<tx::transaction_cid_t>(main_tables[0]->size(), tx::INF_CID);
63  _tidVector = std::vector<tx::transaction_id_t>(main_tables[0]->size(), tx::START_TID);
64 
65  // Replace the delta partition
66  delta = new_delta;
67 }

253 bool Store::isVisibleForTransaction(pos_t pos, tx::transaction_cid_t last_commit_id, tx::transaction_id_t tid) const {
254  if (_tidVector[pos] == tid) {
255  if (last_commit_id >= _cidBeginVector[pos]) {
256  // row was inserted and committed by another transaction, then deleted by our transaction
257  // if we have a lock for it but someone else committed a delete, something is wrong
258  assert(_cidEndVector[pos] == tx::INF_CID);
259  return false;
260  } else {
261  // we inserted this row - nobody should have deleted it yet
262  assert(_cidEndVector[pos] == tx::INF_CID);
263  return true;
264  }
265  } else {
266  if (last_commit_id >= _cidBeginVector[pos]) {
267  // we are looking at a row that was inserted and deleted before we started - we should see it unless it was already deleted again
268  if(last_commit_id >= _cidEndVector[pos]) {
269  // the row was deleted and the delete was committed before we started our transaction
270  return false;
271  } else {
272  // the row was deleted but the commit happened after we started our transaction (or the delete was not committed yet)
273  return true;
274  }
275  } else {
276  // we are looking at a row that was inserted after we started
277  assert(_cidEndVector[pos] > last_commit_id);
278  return false;
279  }
280  }
281 }

311 std::pair<size_t, size_t> Store::appendToDelta(size_t num) {
312  static locking::Spinlock mtx;
313  std::lock_guard<locking::Spinlock> lck(mtx);
314 
315  size_t start = delta->size();
316  std::pair<size_t, size_t> result = {start, start + num};
317 
318  // Update Delta
319  delta->resize(start + num);
320  // Update CID, TID and valid
321  auto main_tables_size = functional::sum(main_tables, 0ul, [](atable_ptr_t& t){return t->size();});
322 
323  _cidBeginVector.resize(main_tables_size + start + num, tx::INF_CID);
324  _cidEndVector.resize(main_tables_size + start + num, tx::INF_CID);
325  _tidVector.resize(main_tables_size + start + num, tx::START_TID);
326 
327  return std::move(result);
328 }
